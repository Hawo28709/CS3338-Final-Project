\section*{Dependencies}
\begin{itemize}
    \item \textbf{Front-End (React Native):}
    \begin{itemize}
        \item Node.js and npm
        \item Expo Go (for testing and deployment)
        \item React Native Navigation Components
        \item AsyncStorage (local database)
    \end{itemize}

    \item \textbf{Back-End (AWS + Cloudflare):}
    \begin{itemize}
        \item AWS Lambda: Handles API logic and chatbot requests
        \item API Gateway: Exposes REST endpoints to the app
        \item Amazon Cognito: Handles user authentication and the creation, updation and deletion of accounts
        \item Cloudflare D1: Stores user data and tests events
    \end{itemize}

    \item \textbf{AI Layer:}
    \begin{itemize}
        \item OpenAI GPT-based chatbot
        \item Custom prompts for promoting mental health and wellness, and other resources
    \end{itemize}

    \item \textbf{Database Management:}
    \begin{itemize}
        \item Cloudflare D1 Console
        \item SQL schema for users, quiz scores, and resource data
    \end{itemize}
\end{itemize}
